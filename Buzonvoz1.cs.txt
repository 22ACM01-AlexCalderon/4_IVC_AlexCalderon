internal class Program
{
    private static void Main(string[] args)
    {
        int n;
        Console.WriteLine("===BUZON DE VOZ===");
        Console.WriteLine("Atencion al cliente: 1");
        Console.WriteLine("Soporte tecnico: 2");
        Console.WriteLine("Ventas: 3");    
        Console.WriteLine("Dejar mensaje: 4");
        Console.WriteLine("Ingrese el numero de la opcion que desea:");
        n = Convert.ToInt32(Console.ReadLine());

        if (n == 1)
        {
            Console.WriteLine("Atencion al cliente seleccionado");
        }
        else if (n == 2)
        {
            Console.WriteLine("En un momento sera atendido por un Tecnico");
        }
        else if (n == 3)
        {
            Console.WriteLine("Bienvenido al departamento de Ventas");
        }
        else if (n == 4)
        {
            Console.WriteLine("Deje su mensaje despues del tono");
        }
    }
}
